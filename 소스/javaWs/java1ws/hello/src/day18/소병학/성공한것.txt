package day18.소병학;import java.util.Arrays;import java.util.Map;public class Theater {    private int[] size;    private Map<String, String[]> movieAndTime;    // Default constructor    public Theater() {    }    // All-argument constructor    public Theater(int[] size, Map<String, String[]> movieAndTime) {        this.size = size;        this.movieAndTime = movieAndTime;    }    public int[] getSize() {        return size;    }    public void setSize(int[] size) {        this.size = size;    }    public Map<String, String[]> getMovieAndTime() {        return movieAndTime;    }    public void setMovieAndTime(Map<String, String[]> movieAndTime) {        this.movieAndTime = movieAndTime;    }    @Override    public String toString() {         return "{ size=" + Arrays.toString(size) + ", movieAndTime=" + movieAndTime + " }";    	//return movieAndTime;    }}
